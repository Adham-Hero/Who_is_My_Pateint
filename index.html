// معالجة تسجيل الدخول document.getElementById('loginForm').onsubmit = async (e) => { e.preventDefault(); const username = document.getElementById('username').value; const password = document.getElementById('password').value; const message = document.getElementById('message'); try { // ملاحظة: قمنا بتغيير الرابط ليعمل على Vercel تلقائياً const res = await fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username, password }) }); // تحويل الرد إلى JSON const result = await res.json(); if (res.ok && result.success) { // تخزين اسم المستخدم والدور في المتصفح لاستخدامهما في الصفحات الأخرى localStorage.setItem('userRole', result.role); // التوجيه بناءً على الدور (Role) القادم من السيرفر if (result.role === 'admin') { window.location.href = 'admin.html'; } else { window.location.href = 'user.html'; } } else { // عرض رسالة الخطأ القادمة من السيرفر message.innerText = result.message || 'اسم المستخدم أو كلمة المرور غير صحيحة'; } } catch (err) { console.error("Login Error:", err); message.innerText = 'تعذر الاتصال بالسيرفر. تأكد من رفع الملفات بشكل صحيح.'; } };
